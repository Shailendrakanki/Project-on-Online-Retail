                 .transform('min') \
                 .dt.to_period('M')
In [33]:
Retails_Cohort = Retails.groupby(['Cohort', 'Order_Month']) \
              .agg(n_customers=('CustomerID', 'nunique')) \
              .reset_index(drop=False)
Retails_Cohort['Period_Number'] = (Retails_Cohort.Order_Month - Retails_Cohort.Cohort).apply(attrgetter('n'))
In [34]:
Cohort_pivot = Retails_Cohort.pivot_table(index = 'Cohort',
                                     columns = 'Period_Number',
                                     values = 'n_customers')
In [35]:
Cohort_size = Cohort_pivot.iloc[:,0]
Retention_Matrix = Cohort_pivot.divide(Cohort_size, axis = 0)
